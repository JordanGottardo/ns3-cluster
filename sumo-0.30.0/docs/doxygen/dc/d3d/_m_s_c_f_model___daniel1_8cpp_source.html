54 double
55 MSCFModel_Daniel1::moveHelper(MSVehicle* const veh, double vPos) const {
56  const double oldV = veh->getSpeed(); // save old v for optional acceleration computation
57  const double vSafe = MIN2(vPos, veh->processNextStop(vPos)); // process stops
58  // we need the acceleration for emission computation;
59  // in this case, we neglect dawdling, nonetheless, using
60  // vSafe does not incorporate speed reduction due to interaction
61  // on lane changing
62  const double vMin = getSpeedAfterMaxDecel(oldV);
63  const double vMax = MIN3(veh->getLane()->getVehicleMaxSpeed(veh), maxNextSpeed(oldV, veh), vSafe);
64 #ifdef _DEBUG
65  if (vMin > vMax) {
66  WRITE_WARNING("Maximum speed of vehicle '" + veh->getID() + "' is lower than the minimum speed (min: " + toString(vMin) + ", max: " + toString(vMax) + ").");
67  }
68 #endif
69  return veh->getLaneChangeModel().patchSpeed(vMin, MAX2(vMin, dawdle(vMax)), vMax, *this);
70 }

92 double MSCFModel_Daniel1::_vsafe(double gap, double predSpeed) const {
93  if (predSpeed == 0 && gap < 0.01) {
94  return 0;
95  }
96  double vsafe = (double)(-1. * myTauDecel
97  + sqrt(
98  myTauDecel * myTauDecel
99  + (predSpeed * predSpeed)
100  + (2. * myDecel * gap)
101  ));
102  assert(vsafe >= 0);
103  return vsafe;
104 }
