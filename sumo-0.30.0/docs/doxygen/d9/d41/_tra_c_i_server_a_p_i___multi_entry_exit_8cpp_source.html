43 bool
44 TraCIServerAPI_MultiEntryExit::processGet(TraCIServer& server, tcpip::Storage& inputStorage,
45  tcpip::Storage& outputStorage) {
46  // variable & id
47  int variable = inputStorage.readUnsignedByte();
48  std::string id = inputStorage.readString();
49  // check variable
50  if (variable != ID_LIST && variable != LAST_STEP_VEHICLE_NUMBER && variable != LAST_STEP_MEAN_SPEED
51  && variable != LAST_STEP_VEHICLE_ID_LIST && variable != LAST_STEP_VEHICLE_HALTING_NUMBER
52  && variable != ID_COUNT) {
53  return server.writeErrorStatusCmd(CMD_GET_MULTIENTRYEXIT_VARIABLE, "Get MeMeDetector Variable: unsupported variable " + toHex(variable, 2) + " specified", outputStorage);
54  }
55  // begin response building
56  tcpip::Storage tempMsg;
57  // response-code, variableID, objectID
58  tempMsg.writeUnsignedByte(RESPONSE_GET_MULTIENTRYEXIT_VARIABLE);
59  tempMsg.writeUnsignedByte(variable);
60  tempMsg.writeString(id);
61  try {
62  switch (variable) {
63  case ID_LIST:
64  tempMsg.writeUnsignedByte(TYPE_STRINGLIST);
65  tempMsg.writeStringList(TraCI_MultiEntryExit::getIDList());
66  break;
67  case ID_COUNT:
68  tempMsg.writeUnsignedByte(TYPE_INTEGER);
69  tempMsg.writeInt(TraCI_MultiEntryExit::getIDCount());
70  break;
71  case LAST_STEP_VEHICLE_NUMBER:
72  tempMsg.writeUnsignedByte(TYPE_INTEGER);
73  tempMsg.writeInt(TraCI_MultiEntryExit::getLastStepVehicleNumber(id));
74  break;
75  case LAST_STEP_MEAN_SPEED:
76  tempMsg.writeUnsignedByte(TYPE_DOUBLE);
77  tempMsg.writeDouble(TraCI_MultiEntryExit::getLastStepMeanSpeed(id));
78  break;
79  case LAST_STEP_VEHICLE_ID_LIST:
80  tempMsg.writeUnsignedByte(TYPE_STRINGLIST);
81  tempMsg.writeStringList(TraCI_MultiEntryExit::getLastStepVehicleIDs(id));
82  break;
83  case LAST_STEP_VEHICLE_HALTING_NUMBER:
84  tempMsg.writeUnsignedByte(TYPE_INTEGER);
85  tempMsg.writeInt(TraCI_MultiEntryExit::getLastStepHaltingNumber(id));
86  break;
87  default:
88  break;
89  }
90  } catch (TraCIException& e) {
91  return server.writeErrorStatusCmd(CMD_GET_MULTIENTRYEXIT_VARIABLE, e.what(), outputStorage);
92  }
93  server.writeStatusCmd(CMD_GET_MULTIENTRYEXIT_VARIABLE, RTYPE_OK, "", outputStorage);
94  server.writeResponseWithLength(outputStorage, tempMsg);
95  return true;
96 }
