96 double
97 MSCFModel_IDM::interactionGap(const MSVehicle* const veh, double vL) const {
98  // Resolve the IDM equation to gap. Assume predecessor has
99  // speed != 0 and that vsafe will be the current speed plus acceleration,
100  // i.e that with this gap there will be no interaction.
101  const double acc = myAccel * (1. - pow(veh->getSpeed() / veh->getLane()->getVehicleMaxSpeed(veh), myDelta));
102  const double vNext = veh->getSpeed() + acc;
103  const double gap = (vNext - vL) * (veh->getSpeed() + vL) / (2 * myDecel) + vL;
104 
105  // Don't allow timeHeadWay < deltaT situations.
106  return MAX2(gap, SPEED2DIST(vNext));
107 }

110 double
111 MSCFModel_IDM::_v(const MSVehicle* const veh, const double gap2pred, const double egoSpeed,
112  const double predSpeed, const double desSpeed, const bool respectMinGap) const {
113 // this is more or less based on http://www.vwi.tu-dresden.de/~treiber/MicroApplet/IDM.html
114 // and http://arxiv.org/abs/cond-mat/0304337
115 // we assume however constant speed for the leader
116  double headwayTime = myHeadwayTime;
117  if (myAdaptationFactor != 1.) {
118  const VehicleVariables* vars = (VehicleVariables*)veh->getCarFollowVariables();
119  headwayTime *= myAdaptationFactor + vars->levelOfService * (1. - myAdaptationFactor);
120  }
121  double newSpeed = egoSpeed;
122  double gap = gap2pred;
123  for (int i = 0; i < myIterations; i++) {
124  const double delta_v = newSpeed - predSpeed;
125  double s = MAX2(0., newSpeed * headwayTime + newSpeed * delta_v / myTwoSqrtAccelDecel);
126  if (respectMinGap) {
127  s += myType->getMinGap();
128  }
129  const double acc = myAccel * (1. - pow(newSpeed / desSpeed, myDelta) - (s * s) / (gap * gap));
130  newSpeed += ACCEL2SPEED(acc) / myIterations;
131  //TODO use more realistic position update which takes accelerated motion into account
132  gap -= MAX2(0., SPEED2DIST(newSpeed - predSpeed) / myIterations);
133  }
134 // return MAX2(getSpeedAfterMaxDecel(egoSpeed), newSpeed);
135  return MAX2(0., newSpeed);
136 }
