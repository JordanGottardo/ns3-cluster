52 NBHelpers::relAngle(double angle1, double angle2) {
53  angle2 -= angle1;
54  if (angle2 > 180) {
55  angle2 = (360 - angle2) * -1;
56  }
57  while (angle2 < -180) {
58  angle2 = 360 + angle2;
59  }
60  return angle2;
61 }
