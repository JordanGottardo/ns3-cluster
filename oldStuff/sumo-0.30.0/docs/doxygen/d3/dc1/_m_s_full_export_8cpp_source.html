63 void
64 MSFullExport::writeVehicles(OutputDevice& of) {
65  of.openTag("vehicles");
66  MSVehicleControl& vc = MSNet::getInstance()->getVehicleControl();
67  for (MSVehicleControl::constVehIt it = vc.loadedVehBegin(); it != vc.loadedVehEnd(); ++it) {
68  const SUMOVehicle* veh = it->second;
69  const MSVehicle* microVeh = dynamic_cast<const MSVehicle*>(veh);
70  if (veh->isOnRoad()) {
71  std::string fclass = veh->getVehicleType().getID();
72  fclass = fclass.substr(0, fclass.find_first_of("@"));
73  PollutantsInterface::Emissions emiss = PollutantsInterface::computeAll(veh->getVehicleType().getEmissionClass(), veh->getSpeed(), veh->getAcceleration(), veh->getSlope());
74  of.openTag("vehicle").writeAttr("id", veh->getID()).writeAttr("eclass", PollutantsInterface::getName(veh->getVehicleType().getEmissionClass()));
75  of.writeAttr("CO2", emiss.CO2).writeAttr("CO", emiss.CO).writeAttr("HC", emiss.HC).writeAttr("NOx", emiss.NOx);
76  of.writeAttr("PMx", emiss.PMx).writeAttr("fuel", emiss.fuel).writeAttr("electricity", emiss.electricity);
77  of.writeAttr("noise", HelpersHarmonoise::computeNoise(veh->getVehicleType().getEmissionClass(), veh->getSpeed(), veh->getAcceleration()));
78  of.writeAttr("route", veh->getRoute().getID()).writeAttr("type", fclass);
79  if (microVeh != 0) {
80  of.writeAttr("waiting", microVeh->getWaitingSeconds());
81  of.writeAttr("lane", microVeh->getLane()->getID());
82  }
83  of.writeAttr("pos", veh->getPositionOnLane()).writeAttr("speed", veh->getSpeed());
84  of.writeAttr("angle", GeomHelper::naviDegree(veh->getAngle())).writeAttr("x", veh->getPosition().x()).writeAttr("y", veh->getPosition().y());
85  of.closeTag();
86  }
87  }
88  of.closeTag();
89 }

91 void
92 MSFullExport::writeEdge(OutputDevice& of) {
93  of.openTag("edges");
94  MSEdgeControl& ec = MSNet::getInstance()->getEdgeControl();
95  const MSEdgeVector& edges = ec.getEdges();
96  for (MSEdgeVector::const_iterator e = edges.begin(); e != edges.end(); ++e) {
97  MSEdge& edge = **e;
98  of.openTag("edge").writeAttr("id", edge.getID()).writeAttr("traveltime", edge.getCurrentTravelTime());
99  const std::vector<MSLane*>& lanes = edge.getLanes();
100  for (std::vector<MSLane*>::const_iterator lane = lanes.begin(); lane != lanes.end(); ++lane) {
101  writeLane(of, **lane);
102  }
103  of.closeTag();
104  }
105  of.closeTag();
106 }

121 void
122 MSFullExport::writeTLS(OutputDevice& of, SUMOTime /* timestep */) {
123  of.openTag("tls");
124  MSTLLogicControl& vc = MSNet::getInstance()->getTLSControl();
125  std::vector<std::string> ids = vc.getAllTLIds();
126  for (std::vector<std::string>::const_iterator id_it = ids.begin(); id_it != ids.end(); ++id_it) {
127  MSTLLogicControl::TLSLogicVariants& vars = MSNet::getInstance()->getTLSControl().get(*id_it);
128  const MSTrafficLightLogic::LaneVectorVector& lanes = vars.getActive()->getLaneVectors();
129 
130  std::vector<std::string> laneIDs;
131  for (MSTrafficLightLogic::LaneVectorVector::const_iterator i = lanes.begin(); i != lanes.end(); ++i) {
132  const MSTrafficLightLogic::LaneVector& llanes = (*i);
133  for (MSTrafficLightLogic::LaneVector::const_iterator j = llanes.begin(); j != llanes.end(); ++j) {
134  laneIDs.push_back((*j)->getID());
135  }
136  }
137 
138  std::string lane_output = "";
139  for (int i1 = 0; i1 < (int)laneIDs.size(); ++i1) {
140  lane_output += laneIDs[i1] + " ";
141  }
142 
143  std::string state = vars.getActive()->getCurrentPhaseDef().getState();
144  of.openTag("trafficlight").writeAttr("id", *id_it).writeAttr("state", state).closeTag();
145  }
146  of.closeTag();
147 }
