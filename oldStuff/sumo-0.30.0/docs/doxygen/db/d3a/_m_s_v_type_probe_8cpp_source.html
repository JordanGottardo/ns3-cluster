62 MSVTypeProbe::execute(SUMOTime currentTime) {
63  myOutputDevice.openTag(SUMO_TAG_TIMESTEP);
64  myOutputDevice.writeAttr(SUMO_ATTR_TIME, time2string(currentTime));
65  myOutputDevice.writeAttr(SUMO_ATTR_ID, getID());
66  myOutputDevice.writeAttr("vType", myVType);
67  MSVehicleControl& vc = MSNet::getInstance()->getVehicleControl();
68  for (MSVehicleControl::constVehIt it = vc.loadedVehBegin(); it != vc.loadedVehEnd(); ++it) {
69  const SUMOVehicle* veh = it->second;
70  const MSVehicle* microVeh = dynamic_cast<const MSVehicle*>(veh);
71  if (myVType == "" || myVType == veh->getVehicleType().getID()) {
72  if (veh->isOnRoad()) {
73  Position pos = veh->getPosition();
74  myOutputDevice.openTag(SUMO_TAG_VEHICLE);
75  myOutputDevice.writeAttr(SUMO_ATTR_ID, veh->getID());
76  if (microVeh != 0) {
77  myOutputDevice.writeAttr(SUMO_ATTR_LANE, microVeh->getLane()->getID());
78  }
79  myOutputDevice.writeAttr(SUMO_ATTR_POSITION, veh->getPositionOnLane());
80  myOutputDevice.writeAttr(SUMO_ATTR_X, pos.x());
81  myOutputDevice.writeAttr(SUMO_ATTR_Y, pos.y());
82  if (MSNet::getInstance()->hasElevation()) {
83  myOutputDevice.writeAttr(SUMO_ATTR_Z, pos.z());
84  }
85  if (GeoConvHelper::getFinal().usingGeoProjection()) {
86  GeoConvHelper::getFinal().cartesian2geo(pos);
87  myOutputDevice.setPrecision(gPrecisionGeo);
88  myOutputDevice.writeAttr(SUMO_ATTR_LAT, pos.y());
89  myOutputDevice.writeAttr(SUMO_ATTR_LON, pos.x());
90  myOutputDevice.setPrecision();
91  }
92  myOutputDevice.writeAttr(SUMO_ATTR_SPEED, veh->getSpeed());
93  myOutputDevice.closeTag();
94  }
95  }
96  }
97  myOutputDevice.closeTag();
98  return myFrequency;
99 }
